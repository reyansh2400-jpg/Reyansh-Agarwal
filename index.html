        // Countdown to January 11, 2026

        const targetDate = new Date('2026-01-11T00:00:00').getTime();


        function updateCountdown() {

            const now = new Date().getTime();

            const distance = targetDate - now;


            const days = Math.floor(distance / (1000 * 60 * 60 * 24));


            document.getElementById('countdown').innerHTML = days + ' days to go!';


            if (distance < 0) {

                document.getElementById('countdown').innerHTML = 'The day has arrived!';

            }

        }


        updateCountdown();

        setInterval(updateCountdown, 1000 * 60 * 60); // Update every hour
